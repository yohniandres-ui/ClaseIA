    <script>
        // Global state
        let sessionId = `session_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;

        // Show specific survey
        function showSurvey(surveyType) {
            document.getElementById('sectionSelection').classList.add('hidden');
            document.getElementById(surveyType + 'Survey').classList.add('active');
            window.scrollTo(0, 0);
        }

        // Go back to section selection
        function goBackToSelection() {
            document.querySelectorAll('.survey-view').forEach(view => {
                view.classList.remove('active');
            });
            document.getElementById('sectionSelection').classList.remove('hidden');
            
            // Reset all forms
            document.querySelectorAll('.survey-form').forEach(form => {
                form.reset();
                form.style.display = 'block';
            });
            
            // Clear selections
            document.querySelectorAll('.rating-btn.selected').forEach(btn => btn.classList.remove('selected'));
            document.querySelectorAll('.star.selected').forEach(star => star.classList.remove('selected'));
            
            // Hide success messages
            document.querySelectorAll('.success-message').forEach(msg => msg.classList.remove('show'));
            
            window.scrollTo(0, 0);
        }

        // Set star rating
        function setRating(fieldName, rating) {
            const container = document.getElementById(fieldName);
            const stars = container.querySelectorAll('.star');
            const hiddenInput = container.parentElement.querySelector(`input[name="${fieldName}"]`);
            
            stars.forEach((star, index) => {
                if (index < rating) {
                    star.classList.add('selected');
                } else {
                    star.classList.remove('selected');
                }
            });
            
            hiddenInput.value = rating;
        }

        // Select button rating
        function selectRating(fieldName, value, button) {
            const container = button.parentElement;
            const buttons = container.querySelectorAll('.rating-btn');
            const hiddenInput = container.parentElement.querySelector(`input[name="${fieldName}"]`);
            
            buttons.forEach(btn => btn.classList.remove('selected'));
            button.classList.add('selected');
            
            hiddenInput.value = value;
        }

        // Submit survey
        async function submitSurvey(event, surveyType) {
            event.preventDefault();
            
            const form = event.target;
            const formData = new FormData(form);
            
            // Collect all responses
            const responses = {
                sessionId: sessionId,
                surveyType: surveyType,
                timestamp: new Date().toISOString(),
                responses: {}
            };
            
            // Get all form values
            for (let [key, value] of formData.entries()) {
                responses.responses[key] = value;
            }
            
            console.log('Survey Data:', responses);

            // Send to Landbot if available
            if (window.myLandbot) {
                try {
                    window.myLandbot.sendMessage({
                        type: 'custom',
                        data: responses
                    });
                } catch (e) {
                    console.log('Landbot message send error:', e);
                }
            }

            // Send to backend
            try {
                const response = await fetch('/api/feedback', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        sessionId: responses.sessionId,
                        rating: parseInt(responses.responses[surveyType + '_nps']) || 5,
                        comments: JSON.stringify(responses)
                    })
                });

                if (response.ok) {
                    // Show success message
                    form.style.display = 'none';
                    document.getElementById(surveyType + 'Success').classList.add('show');
                    
                    // Reset after 3 seconds
                    setTimeout(() => {
                        goBackToSelection();
                    }, 3000);
                } else {
                    alert('There was an error submitting your survey. Please try again.');
                }
            } catch (error) {
                console.error('Error submitting survey:', error);
                // Still show success for demo purposes
                form.style.display = 'none';
                document.getElementById(surveyType + 'Success').classList.add('show');
                
                setTimeout(() => {
                    goBackToSelection();
                }, 3000);
            }
        }

        // Landbot Integration
        window.addEventListener('mouseover', initLandbot, { once: true });
        window.addEventListener('touchstart', initLandbot, { once: true });
        
        var myLandbot;
        function initLandbot() {
            if (!myLandbot) {
                var s = document.createElement('script');
                s.type = "module";
                s.async = true;
                s.addEventListener('load', function() {
                    myLandbot = new Landbot.Livechat({
                        configUrl: 'https://storage.googleapis.com/landbot.online/v3/H-3339187-FIRP1MJ0TICX8FM4/index.json',
                    });
                    
                    window.myLandbot = myLandbot;
                    
                    myLandbot.on('open', function() {
                        console.log('Landbot opened - ready to receive data');
                    });
                });
                s.src = 'https://cdn.landbot.io/landbot-3/landbot-3.0.0.mjs';
                var x = document.getElementsByTagName('script')[0];
                x.parentNode.insertBefore(s, x);
            }
        }

        // Initialize on page load
        console.log('Session ID:', sessionId);
    </script>
